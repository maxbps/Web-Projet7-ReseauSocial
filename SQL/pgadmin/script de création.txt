
CREATE TABLE public.Users (
                user_email VARCHAR NOT NULL,
                user_name VARCHAR NOT NULL,
                user_psw VARCHAR NOT NULL,
                user_picture VARCHAR NOT NULL,
                CONSTRAINT users_pk PRIMARY KEY (user_email)
);


CREATE SEQUENCE public.posts_post_id_seq;

CREATE TABLE public.Posts (
                post_id INTEGER NOT NULL DEFAULT nextval('public.posts_post_id_seq'),
                user_email VARCHAR NOT NULL,
                post_description VARCHAR NOT NULL,
                post_picture VARCHAR NOT NULL,
                CONSTRAINT posts_pk PRIMARY KEY (post_id)
);


ALTER SEQUENCE public.posts_post_id_seq OWNED BY public.Posts.post_id;

ALTER TABLE public.Posts ADD CONSTRAINT user_post_fk
FOREIGN KEY (user_email)
REFERENCES public.Users (user_email)
ON DELETE NO ACTION
ON UPDATE NO ACTION
NOT DEFERRABLE;
